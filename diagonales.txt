DEBUT diagonales
	VARIABLES LOCALES
		CONSTANTE LIGNES:N <- 4
		CONSTANTE COLONES:N <- 4
		var tab:N [LIGNES,COLONES]
		var i:N <- 1
		var j:N <- 1
	FIN VARIABLES LOCALES

	POUR i ALLANT DE 1 A LIGNES
		j <- 1
		POUR j ALLANT DE 1 A COLONES
			SI i=j
			tab[i,j] <- 1
			FIN SI

			SI i+j=5
			tab[i,j] <-2

		j SUIVANT

	i SUIVANT

FIN diagonales
