DEBUT dichotomique
	VARIABLES LOCALES
		CONSTANTE TAILLE:N <- 1000
		VAR tab:N[TAILLE]
		VAR n:N <- 1
		VAR i:N <- 1
		VAR debut:N <- 1
		VAR fin:N <- TAILLE
	FIN VARIABLES LOCALES

	// ici, valeur de N déterminée d'une manière ou d'une autre

	TANT QUE n != tab[i] ET debut != FIN
		i <- (debut + fin) DIV 2

		SI n > tab[i] 
			debut <- i

		SINON
			fin <- i

	FIN TANT QUE

	SI n = tab [i]
		AFFICHER n, "est placé à la position", tab[i]
	SINON
		AFFICHER "valeur non trouvée"
	FIN SI


FIN dichotomique
